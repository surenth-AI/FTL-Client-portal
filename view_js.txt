        function setDefaultCargoReadyDate() {
            const today = new Date();
            const yyyy = today.getFullYear();
            const mm = String(today.getMonth() + 1).padStart(2, '0');
            const dd = String(today.getDate()).padStart(2, '0');
            const todayString = `${yyyy}-${mm}-${dd}`;

            const dateInput = document.getElementById('cargoReadyDate');
            if (dateInput) {
                dateInput.value = todayString;
                dateInput.min = todayString; // Prevent selecting past dates
            }
        }

            selectedLocations.dest = search.destData.location;
            document.getElementById('destLocation').value = `${search.destData.location.name} (${search.destData.location.code})`;

            console.log('Loaded recent search:', search);

        }

        function togglePanel(panelName) {
            console.log('togglePanel called with:', panelName);
            const panels = ['route', 'cargo', 'services'];
            const panel = document.getElementById(panelName + 'Panel');
            const section = document.getElementById(panelName + 'Section');

            console.log('panel element:', panel);
            console.log('section element:', section);

            panels.forEach(p => {
                const otherPanel = document.getElementById(p + 'Panel');
                const otherSection = document.getElementById(p + 'Section');
                if (otherPanel && p !== panelName) {
                    otherPanel.classList.remove('open');
                    otherSection.classList.remove('active');
                }
            });

            if (panel) {
                const isOpen = panel.classList.contains('open');
                console.log('Panel isOpen:', isOpen);
                if (isOpen) {
                    panel.classList.remove('open');
                    section.classList.remove('active');
                    currentPanel = null;
                } else {
                    panel.classList.add('open');
                    section.classList.add('active');
                    currentPanel = panelName;
                    console.log('Panel opened, classes:', panel.className);
                }
            } else {
                console.log('Panel not found!');
            }
        }

        function filterLocations(locationType) {
            const country = document.getElementById(locationType + 'Country').value;
            const type = document.getElementById(locationType + 'Type').value;
            const searchTerm = document.getElementById(locationType + 'Location').value.toLowerCase();
            const dropdown = document.getElementById(locationType + 'Dropdown');

            if (!country) {
                dropdown.classList.remove('show');
                return;
            }

            const dataKey = type === 'door' ? 'cities' : 'ports';
            const locations = locationData[dataKey][country] || [];

            const filtered = locations.filter(loc => {
                if (type === 'door') {
                    return loc.name.toLowerCase().includes(searchTerm) ||
                        loc.zipcode.toLowerCase().includes(searchTerm);
                } else {
                    return loc.name.toLowerCase().includes(searchTerm) ||
                        loc.uncode.toLowerCase().includes(searchTerm);
                }
            });

            if (filtered.length === 0) {
                dropdown.innerHTML = '<div class="dropdown-no-results">No results found</div>';
            } else {
                dropdown.innerHTML = filtered.map(loc => {
                    const code = type === 'door' ? loc.zipcode : loc.uncode;
                    return `
                        <div class="dropdown-item" onclick="selectLocation('${locationType}', '${loc.name}', '${code}')">
                            <div class="dropdown-item-name">${loc.name}</div>
                            <div class="dropdown-item-code">${code}</div>
                        </div>
                    `;
                }).join('');
            }

            dropdown.classList.add('show');
        }

        function updateAvailableIncoterms() {
            const originType = document.getElementById('originType').value;
            const destType = document.getElementById('destType').value;
            const incotermSelect = document.getElementById('incoterm');

            // Clear existing options except the first placeholder
            incotermSelect.innerHTML = '<option value="">Select incoterm</option>';

            // Always available options
            const alwaysAvailable = [
                { value: 'FCA', text: 'FCA - Free Carrier' },
                { value: 'FOB', text: 'FOB - Free On Board' },
                { value: 'CFR', text: 'CFR - Cost and Freight' }
            ];

            alwaysAvailable.forEach(opt => {
                const option = document.createElement('option');
                option.value = opt.value;
                option.textContent = opt.text;
                incotermSelect.appendChild(option);

                const option = document.createElement('option');
                option.value = 'EXW';
                option.textContent = 'EXW - Ex Works';
                incotermSelect.appendChild(option);

            if (destType === 'door') {
                const dapOption = document.createElement('option');
                dapOption.value = 'DAP';
                dapOption.textContent = 'DAP - Delivered At Place';

                dpuOption.value = 'DPU';

        function setUnit(cargoItemId, unitSystem) {
            const buttons = document.querySelectorAll(`#${cargoItemId} .unit-option`);
            buttons.forEach(btn => btn.classList.remove('active'));
            event.target.classList.add('active');

            // Update main cargo field placeholders
            const weightInput = document.getElementById(`${cargoItemId}-weight`);
            const volumeInput = document.getElementById(`${cargoItemId}-volume`);

            if (unitSystem === 'metric') {
                weightInput.placeholder = '0.00 kg';
                volumeInput.placeholder = '0.00 m³';
            } else {
                weightInput.placeholder = '0.00 lbs';
                volumeInput.placeholder = '0.00 ft³';
            }

            // Update all dimension labels and placeholders
            const dimsContainer = document.getElementById(`${cargoItemId}-dimensions`);
            const dimItems = dimsContainer.querySelectorAll('.dimension-item');

            const lengthUnit = unitSystem === 'metric' ? 'cm' : 'in';
            const weightUnit = unitSystem === 'metric' ? 'kg' : 'lbs';

            dimItems.forEach(item => {
                const labels = item.querySelectorAll('.dimension-input-small label');
                const inputs = item.querySelectorAll('.dimension-input-small input');

                // Update length, width, height labels
                if (labels[1]) labels[1].textContent = `Length (${lengthUnit})`;
                if (labels[2]) labels[2].textContent = `Width (${lengthUnit})`;
                if (labels[3]) labels[3].textContent = `Height (${lengthUnit})`;

                // Update placeholders
                if (inputs[1]) inputs[1].placeholder = `0 ${lengthUnit}`;
                if (inputs[2]) inputs[2].placeholder = `0 ${lengthUnit}`;
                if (inputs[3]) inputs[3].placeholder = `0 ${lengthUnit}`;

                // Update weight placeholder
                const weightInput = item.querySelector('[id$="-weight"]');
                if (weightInput) {
                    weightInput.placeholder = `0.00 ${weightUnit}`;
                }
            });

            // Recalculate with new units
            updateTotals(cargoItemId);
        }

        function toggleStackable(cargoItemId) {
            const isStackable = document.getElementById(`${cargoItemId}-stackable`).checked;
            const dimsRequired = document.getElementById(`${cargoItemId}-dims-required`);

            if (!isStackable) {
                dimsRequired.style.display = 'inline';
                // Add at least one dimension if none exist
                const dimsContainer = document.getElementById(`${cargoItemId}-dimensions`);
                if (dimsContainer.children.length === 0) {
                    addDimension(cargoItemId);
                }
            } else {
                dimsRequired.style.display = 'none';
            }
        }

        function updateTotals(cargoItemId) {
            // Get unit system
            const unitButtons = document.querySelectorAll(`#${cargoItemId} .unit-option`);
            let isMetric = true;
            unitButtons.forEach(btn => {
                if (btn.classList.contains('active')) {
                    isMetric = btn.textContent.includes('Metric');
                }
            });

            // Calculate totals from dimensions
            const dimsContainer = document.getElementById(`${cargoItemId}-dimensions`);
            const dimItems = dimsContainer.querySelectorAll('.dimension-item');

            let totalVolume = 0;
            let totalWeight = 0;

            dimItems.forEach(item => {
                const inputs = item.querySelectorAll('input');
                const pieces = parseFloat(inputs[0].value) || 0;
                const length = parseFloat(inputs[1].value) || 0;
                const width = parseFloat(inputs[2].value) || 0;
                const height = parseFloat(inputs[3].value) || 0;
                const weight = parseFloat(inputs[4].value) || 0;

                // Determine if weight is total or per piece
                const dimId = item.id;
                const weightButtons = item.querySelectorAll('.dimension-weight-toggle button');
                let isPerPiece = false;
                weightButtons.forEach((btn, idx) => {
                    if (btn.classList.contains('active') && idx === 1) {
                        isPerPiece = true;
                    }
                });

                // Calculate volume for this dimension line
                if (length && width && height && pieces) {
                    let lineVolume;
                    if (isMetric) {
                        // Convert cm to m³
                        lineVolume = (length * width * height) / 1000000 * pieces;
                    } else {
                        // Convert inches to ft³
                        lineVolume = (length * width * height) / 1728 * pieces;
                    }
                    totalVolume += lineVolume;
                }

                // Calculate weight for this dimension line
                if (weight) {
                    if (isPerPiece) {
                        totalWeight += weight * pieces;
                    } else {
                        totalWeight += weight;
                    }
                }
            });

            // Update total volume if calculated from dimensions
            if (totalVolume > 0) {
                const volumeInput = document.getElementById(`${cargoItemId}-volume`);
                volumeInput.value = totalVolume.toFixed(3);
            }

            // Update total weight if calculated from dimensions
            if (totalWeight > 0) {
                const weightInput = document.getElementById(`${cargoItemId}-weight`);
                weightInput.value = totalWeight.toFixed(2);
            }
        }

        function saveCargo() {
            console.log('saveCargo called');

            // Check if FCL is selected
            if (selectedCargoType === 'fcl') {
                alert('Please contact our office for FCL quotes');
                return;
            }

            const cargoItems = document.querySelectorAll('.cargo-item');
            console.log('Found cargo items:', cargoItems.length);

            if (cargoItems.length === 0) {
                alert('Please add at least one cargo item');
                return;
            }

            // Validate non-stackable items have dimensions with filled values
            let valid = true;
            let errorMessage = '';

            for (let index = 0; index < cargoItems.length; index++) {
                const item = cargoItems[index];
                const stackableCheckbox = item.querySelector('input[id$="-stackable"]');
                const isStackable = stackableCheckbox ? stackableCheckbox.checked : true;

                console.log(`Cargo Item #${index + 1}: isStackable =`, isStackable);

                const dimsContainer = item.querySelector('[id$="-dimensions"]');
                console.log(`Cargo Item #${index + 1}: dimsContainer found =`, !!dimsContainer);
                console.log(`Cargo Item #${index + 1}: dimensions count =`, dimsContainer ? dimsContainer.children.length : 0);

                if (!isStackable) {
                    // Check if at least one dimension entry exists
                    if (!dimsContainer || dimsContainer.children.length === 0) {
                        valid = false;
                        errorMessage = `Cargo Item #${index + 1}: Non-stackable cargo requires dimensions. Please add at least one dimension entry.`;
                        console.log('Validation failed: No dimensions for non-stackable cargo');
                        break;
                    }

                    // Check that all dimension fields are filled in
                    const dimItems = dimsContainer.querySelectorAll('.dimension-item');
                    console.log(`Cargo Item #${index + 1}: dimension items found =`, dimItems.length);

                    for (let dimIndex = 0; dimIndex < dimItems.length; dimIndex++) {
                        const dimItem = dimItems[dimIndex];
                        const inputs = dimItem.querySelectorAll('input[type="number"]');
                        console.log(`Dimension #${dimIndex + 1}: inputs found =`, inputs.length);

                        let emptyFields = [];

                        // Check pieces (first input)
                        if (!inputs[0]?.value || parseFloat(inputs[0].value) <= 0) {
                            emptyFields.push('Pieces');
                        }
                        // Check length (second input)
                        if (!inputs[1]?.value || parseFloat(inputs[1].value) <= 0) {
                            emptyFields.push('Length');
                        }
                        // Check width (third input)
                        if (!inputs[2]?.value || parseFloat(inputs[2].value) <= 0) {
                            emptyFields.push('Width');
                        }
                        // Check height (fourth input)
                        if (!inputs[3]?.value || parseFloat(inputs[3].value) <= 0) {
                            emptyFields.push('Height');
                        }
                        // Check weight (last input with id ending in -weight)
                        const weightInput = dimItem.querySelector('input[id$="-weight"]');
                        if (!weightInput?.value || parseFloat(weightInput.value) < 0) {
                            emptyFields.push('Weight');
                        }

                        console.log(`Dimension #${dimIndex + 1}: emptyFields =`, emptyFields);

                        if (emptyFields.length > 0) {
                            valid = false;
                            errorMessage = `Cargo Item #${index + 1}, Dimension #${dimIndex + 1}: Please fill in all required fields:\n${emptyFields.join(', ')}`;
                            console.log('Validation failed:', errorMessage);
                            break;
                        }
                    }

                    if (!valid) break;
                }
            }

            console.log('Validation result: valid =', valid);
            console.log('Error message:', errorMessage);

            if (!valid) {
                alert(errorMessage);
                return;
            }

            // Calculate totals
            let totalPieces = 0;
            let totalWeight = 0;
            let totalVolume = 0;
            let unit = 'kg/m³'; // Default to metric

            cargoItems.forEach(item => {
                const itemId = item.id;
                const pieces = parseFloat(document.getElementById(`${itemId}-pieces`)?.value) || 0;
                const weight = parseFloat(document.getElementById(`${itemId}-weight`)?.value) || 0;
                const volume = parseFloat(document.getElementById(`${itemId}-volume`)?.value) || 0;

                totalPieces += pieces;
                totalWeight += weight;
                totalVolume += volume;

                // Check unit system
                const unitButtons = item.querySelectorAll('.unit-option');
                unitButtons.forEach(btn => {
                    if (btn.classList.contains('active')) {
                        unit = btn.textContent.includes('Metric') ? 'kg/m³' : 'lbs/ft³';
                    }
                });
            });

            const summary = `${totalPieces} pcs | ${totalWeight.toFixed(2)} ${unit.split('/')[0]} | ${totalVolume.toFixed(3)} ${unit.split('/')[1]}`;
            document.getElementById('cargoValue').textContent = summary;
            document.getElementById('cargoValue').classList.remove('section-placeholder');
            document.getElementById('cargoValue').classList.add('section-value');
            document.getElementById('cargoSection').classList.add('completed');

            togglePanel('cargo');
            setTimeout(() => togglePanel('services'), 300);
        }

        function toggleService(card) {

        function updateChargeInclusions() {
            const incoterm = document.getElementById('incoterm').value;

            if (!incoterm) {
                return;
            }

            const freightCheckbox = document.getElementById('includeFreight');
            const originCheckbox = document.getElementById('includeOriginCharges');
            const destCheckbox = document.getElementById('includeDestCharges');

            // Reset to enabled state
            freightCheckbox.disabled = false;
            originCheckbox.disabled = false;
            destCheckbox.disabled = false;

            let freightIncluded = true;
            let originIncluded = true;
            let destIncluded = false;

            // Set defaults based on incoterm
            switch (incoterm) {
                case 'EXW':
                    // Ex Works: minimal - only origin
                    freightIncluded = false;
                    originIncluded = true;
                    destIncluded = false;
                    freightCheckbox.disabled = true;
                    destCheckbox.disabled = true;
                    break;
                case 'FCA':
                    // Free Carrier: origin + freight
                    freightIncluded = true;
                    originIncluded = true;
                    destIncluded = false;
                    break;
                case 'FOB':
                    // Free On Board: origin + freight
                    freightIncluded = true;
                    originIncluded = true;
                    destIncluded = false;
                    break;
                case 'CFR':
                    // Cost and Freight: origin + freight + dest
                    freightIncluded = true;
                    originIncluded = true;
                    destIncluded = true;
                    break;
                case 'DAP':
                case 'DPU':
                    // Delivered: everything included
                    freightIncluded = true;
                    originIncluded = true;
                    destIncluded = true;
                    break;
            }

            // Apply the settings
            freightCheckbox.checked = freightIncluded;
            originCheckbox.checked = originIncluded;
            destCheckbox.checked = destIncluded;

            // Update visual status
            updateChargeStatus('freight');
            updateChargeStatus('origin');
            updateChargeStatus('dest');
        }

                } else {
                    statusSpan.textContent = 'Excluded';
                    statusSpan.classList.remove('included');
                    statusSpan.classList.add('excluded');
                }
            }
        }

        function showServiceDropdown(type) {
            // Determine which country to use
            let country;
            if (type === 'pickup') {
                country = document.getElementById('originCountry').value;
            } else if (type === 'delivery') {
                country = document.getElementById('destCountry').value;
            }

            if (!country) {
                alert('Please complete the route section first');
                return;
            }

            filterServiceLocations(type);
        }

        function filterServiceLocations(type) {
            let country, searchTerm, dropdown;

            if (type === 'pickup') {
                country = document.getElementById('originCountry').value;
                searchTerm = document.getElementById('pickupLocation').value.toLowerCase();
                dropdown = document.getElementById('pickupDropdown');
            } else if (type === 'delivery') {
                country = document.getElementById('destCountry').value;
                searchTerm = document.getElementById('deliveryLocation').value.toLowerCase();
                dropdown = document.getElementById('deliveryDropdown');
            }

            if (!country) {
                dropdown.classList.remove('show');
                return;
            }

            const cities = locationData.cities[country] || [];

            const filtered = cities.filter(loc =>
                loc.name.toLowerCase().includes(searchTerm) ||
                loc.zipcode.toLowerCase().includes(searchTerm)
            );

            if (filtered.length === 0) {
                dropdown.innerHTML = '<div class="dropdown-no-results">No results found</div>';
            } else {
                dropdown.innerHTML = filtered.map(loc => `
                    <div class="dropdown-item" onclick="selectServiceLocation('${type}', '${loc.name}', '${loc.zipcode}')">
                        <div class="dropdown-item-name">${loc.name}</div>
                        <div class="dropdown-item-code">${loc.zipcode}</div>
                    </div>
                `).join('');
            }

            dropdown.classList.add('show');
        }

        function selectServiceLocation(type, name, code) {

                document.getElementById('pickupLocation').value = `${name} (${code})`;

            } else if (type === 'delivery') {
                document.getElementById('deliveryLocation').value = `${name} (${code})`;
                document.getElementById('deliveryDropdown').classList.remove('show');
            }
        }
